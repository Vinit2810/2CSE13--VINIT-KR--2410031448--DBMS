Query 1: Display empno, ename, deptno, from employee table. Instead of display department numbers display the related department name(Use decode function).

         SELECT EMPNO, ENAME,
         CASE DEPTNO
         WHEN 10 THEN 'RESEARCH'
         WHEN 20 THEN 'ACCOUNTING'
         WHEN 30 THEN 'SALES'
         WHEN 40 THEN 'OPERATIONS'
         END AS DEPT_NAME
         FROMR EMPLOYEE;

Query 2: Display your age in days.
 
         SELECT DATEDIFF(CURDATE(), "2005-10-28");

Query 3: Display your age in months.

         SELECT TIMESTAMPDIFF(MONTH, "2005-10-28", CURDATE()) AS
         DOB_IN_MONTH;

Query 4: Display the current date as 15th August Friday 1997.

Query 7: Find the date for nearest Saturday after current date.

         SELECT DATE_ADD(
         CURDATE(),    
         INTERVAL (5-WEEKDAY(CURDATE())) DAY
         ) AS NEXT_SATURDAY;

Query 8: Display current time.

         SELECT CURTIME();

Query 9: Display the date three months before the current date.

         SELECT DATE_SUB(CURDATE(), INTERVAL 3 MONTH)
         AS DATE_BEFORE_3_MONTHS;

Query 10: Display those employees who joined in the company in the month of Dec.

          SELECT * FROM EMPLOYEE
          WHERE MONTH(HIREDATE) = 12;

Query 11: Display those employees whose first 2 characters from hiredate last 2 characters of salary.

          SELECT * FROM EMPLOYEE
          WHERE LEFT(DATE_FORMAT(HIREDATE, "%Y"),2) = RIGHT(SAL,2);

Query 12: Display those employees whose 10% of salary is equal to the year of joining.

          SELECT * FROM EMPLOYEE
          WHERE (SAL * 0.10) = YEAR(HIREDATE);

Query 13: Display those employees who joined the company before 15th of the months.

          SELECT * FROM EMPLOYEE
          WHERE DAY(HIREDATE) < 15;

Query 14: Display those employees who has joined after 15th of the month.
